provides a set of helper functions that use jQuery * for obtaining values from the inputs of a form. */ var FormHandler = function () { } /** * Returns the value of a field with the specified selector. * * @param {string} selector The input selector. * @returns {string} A string representing the value of the given input. */ FormHandler.prototype.getValue = function (selector) { return $(selector).val() } /** * Returns the values of all fields with the specified selector. * * @param {string} selector The input selector. * @returns {array} An array of strings representing the value for each field * with the given selector. */ FormHandler.prototype.getValues = function (selector) { var inputValues = [] $(selector).each(function () { inputValues.push($(this).val()) }) return inputValues } /** * Returns an array of values of checked inputs. * * @param {string} name The name attribute of the checkboxes. * @returns {array} An array of values of checked inputs. */ FormHandler.prototype.getCheckedBoxes = function (name) { var selector = 'input[name="' + name + '"]:checked' var checked = [] $(selector).each(function () { checked.push($(this).val()) }) return checked } /** * Returns an array of booleans representing the status of each checkbox * with the given name. * * @param {string} name The name attribute of the checkboxes. * @returns {array} An array of booleans. */ FormHandler.prototype.getCheckboxStatus = function (name) { var selector = 'input[name="' + name + '"]' var status = [] $(selector).each(function () { status.push($(this).prop('checked')) }) return status }
