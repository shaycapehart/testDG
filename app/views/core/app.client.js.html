$(document).ready(function () { initialize() // Initialize dropdowns and sidebar, which toggles on mobile displays $('.sidenav').sidenav() $('.dropdown-trigger').dropdown() // Initialize datepickers on dynamically-generated input elements $('body').on('focus', '.datepicker', function () { $(this).datepicker({ yearRange: [1928, (new Date()).getFullYear()], onSelect: function (selectedDate) { var () { var $selector = $(this).parents('.selector'), $checked = $selector.find(':checked'), $reason = $selector.parent().find('input[name^="reason-"]') // Textarea is only used for 'excused' absence type if ($checked.val() === 'E') { $reason.prop({ 'disabled': false, 'required': true }) } else { $reason.prop({ 'disabled': true, 'required': false }).removeClass('valid invalid') } }) // Handle the hover effect on the dynamically-generated Settings page by // revealing help text when the user hovers over a single setting. $(document).on('mouseenter mouseleave', '.setting', function () { $(this).toggleClass('hovered') }) // Handle the click effect on the dynamically-generated Settings page by // giving focus to the input element when the user clicks any part of a // settin. $(document).on('click', '.setting', function () { $(this).find('input').focus() }) }) /** * Load the app by getting the dashboard page. */ function initialize() { getPage('dashboard') } /** * Shows the given page on the app display by running a server-side controller. * * @param {String} page The name of the page to display. * @param {Object} args An optional object for passing controller options. */ function getPage(page, args) { showLoading() google.script.run .withSuccessHandler(updateDisplay) .getAppPage(page, args) } /** * Handles the submit button click by processing the given form data using the * page controller. */ function submit() { var page = $('#submit').data('page') showLoading() switch (page) { case 'addMember': submitMember_() break case 'editAccount': submitAttendance_() break } } /** * Calls the server-side function to gather the chart data and displays the * charts when the data is returned. */ function drawCharts() { google.script.run */ function submitAccount_() { var form = new FormHandler() var account = { rosterId: form.getValue('#rosterId'), fundraiserAmountCheckedOut: form.getValue('#fundraiserAmountCheckedOut'), .saveShirts(shirts) } /** * Constructs the chart displays for the dashboard. * * @param charts The chart data to display. */ function showCharts_(charts) { try { for (var chartName in charts) { var /** * Returns the Google Chart object for the given chart type that will be * displayed in the given target. * * @param {String} type The type of chart to return. * @param {String} target A selector for the target element where the chart * will be displayed. */ function getChartByType_(type, target) { var element = document.getElementById(target) switch (type) { case 'pieChart': return new google.visualization.PieChart(element) case 'barChart': return new google.charts.Bar(element) } } google.charts.load('current', { packages: ['corechart', 'bar'] })
